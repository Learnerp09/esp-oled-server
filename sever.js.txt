const express = require("express");
const app = express();

let text = "Hello Parth 🚀";

app.use(express.urlencoded({ extended: true }));
app.use(express.json());

// Stylish webpage
app.get("/", (req, res) => {
  res.send(`
    <!DOCTYPE html>
    <html>
    <head>
      <title>ESP8266 OLED Control</title>
      <style>
        body {
          font-family: Arial;
          text-align: center;
          background: #0f172a;
          color: white;
          padding-top: 50px;
        }
        input {
          padding: 10px;
          width: 250px;
          border-radius: 10px;
          border: none;
        }
        button {
          padding: 10px 20px;
          border: none;
          border-radius: 10px;
          background: #22c55e;
          color: white;
          font-size: 16px;
          cursor: pointer;
        }
        button:hover {
          background: #16a34a;
        }
      </style>
    </head>
    <body>
      <h1>📟 OLED Text Controller</h1>
      <form action="/set" method="POST">
        <input type="text" name="msg" placeholder="Enter text" required />
        <br><br>
        <button type="submit">Update Display</button>
      </form>
      <p>Current Text: <b>${text}</b></p>
    </body>
    </html>
  `);
});

// API for ESP8266
app.get("/text", (req, res) => {
  res.send(text);
});

// Update text
app.post("/set", (req, res) => {
  text = req.body.msg;
  res.redirect("/");
});

app.listen(3000, () => console.log("Server running"));
